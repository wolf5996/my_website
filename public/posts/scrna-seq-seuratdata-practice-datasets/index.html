Normalization and Clustering Link to heading 

# Test different approaches instantly
pbmc3k <- NormalizeData(pbmc3k) %>%
  FindVariableFeatures() %>%
  ScaleData() %>%
  RunPCA() %>%
  FindNeighbors() %>%
  FindClusters(resolution = 0.5) %>%
  RunUMAP(dims = 1:10)

DimPlot(pbmc3k, reduction = "umap")
